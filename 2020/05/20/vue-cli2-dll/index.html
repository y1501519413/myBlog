# Vue打包优化 --提取vendor

公用库提取，后续打包不再打包这些库

::: tip Front matter variables are optional in VuePress. :::

# 安装依赖

npm install clean-webpack-plugin@^1.0.0 add-asset-html-webpack-plugin@2.1.0 --D

# build 目录下创建 webpack.dll.conf.js

// 处理第三方库，先行打包第三方库，告诉webpack后续不再处理
const webpack = require('webpack')
const path = require('path')
const CleanWebpackPlugin = require('clean-webpack-plugin')
const dllPath = path.resolve(__dirname, '../src/assets/dll') // dll文件存放的目录

module.exports = {
  entry: { // 把 vue 相关模块的放到一个单独的动态链接库
    vue: ['axios', 'vue', 'vue-router', 'vuex', 'element-ui','NProgress']
  },
  output: {
    filename: '[name]-[hash].dll.js', // 生成vue.dll.js
    path: dllPath, // 输出路径
    library: '_dll_[name]' // 证书名称
  },
  plugins: [
    new CleanWebpackPlugin(['*.js'], { // 清除已存在的dll文件
      root: dllPath
    }),
    new webpack.DefinePlugin({
      'process.env': {
        NODE_ENV: 'production' // 设置环境变量
      }
    }),
    new webpack.DllPlugin({
      name: '_dll_[name]', // manifest.json 描述动态链接库包含了哪些内容
      path: path.join(__dirname, './', '[name].dll.manifest.json')
    }),
    // 压缩代码
    new webpack.optimize.UglifyJsPlugin({
      compress: {
        warnings: false,
        pure_funcs: ['console.log']
      },
      sourceMap: false
    })
  ]
}

# 在 package.json 中新增 dll 构建命令

"scripts": {    
  "dll": "webpack --config build/webpack.dll.conf.js" // dll打包命令
},

# 修改 build/webpack.prod.conf.js

const AddAssetHtmlPlugin = require('add-asset-html-webpack-plugin')
plugins: [    
  // 引用 manifest.json
  new webpack.DllReferencePlugin({
    manifest: require('./vue.dll.manifest.json')
  }),    
  // 将 dll 注入到 生成的 html 模板中
  new AddAssetHtmlPlugin({
    filepath: path.resolve(__dirname, '../src/assets/dll/*.js'), // dll文件位置
    publicPath: config.build.assetsPublicPath + utils.assetsPath('dll/'), // dll 引用路径
    outputPath: config.build.assetsPublicPath + utils.assetsPath('dll/'), // dll最终输出的目录
    includeSourcemap: false
    // hash: true,
  }),
]
